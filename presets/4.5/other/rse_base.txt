### Features and Config

#beeper -ARMING_GPS_FIX
#beeper -GPS_STATUS

beeper -SYSTEM_INIT
beeper -ON_USB
beeper -CRASH_FLIP
beeper -CAM_CONNECTION_OPEN
beeper -CAM_CONNECTION_CLOSE
beeper -RC_SMOOTHING_INIT_FAIL

# beacon
beacon RX_LOST
beacon RX_SET


### Radio and Control

# TX/RX
set min_check = 1010
set max_check = 1990
set serialrx_provider = CRSF
set rssi_channel = 15				# ELRS has LQ on 15 and dB on 16, CRSF on 12
rxfail 0 s 1500					# Set Roll rate to 0 (1500)
rxfail 1 s 1500					# Set Pitch rate to 0 (1500)
rxfail 2 s 1500					# Set Yaw rate to 0 (1500)
rxfail 3 s 1250					# Set Throttle to 25% (1250)
rxfail 5 h					# Hold Aux 2 (Prearm / Beep)
rxfail 7 s 1000					# Set Aux 4 (Acro / Angle) to Angle (1000)
rxfail 15 s 1000				# Set RSSI/LQ to zero

#$ OPTION_GROUP BEGIN: Link
  #$ OPTION BEGIN (UNCHECKED): ELRS
    set rssi_channel = 15			# ELRS has LQ on 15 and dB on 16, CRSF on 12
    rxfail 15 s 1000				# Set RSSI/LQ to zero
  #$ OPTION END
  #$ OPTION BEGIN (UNCHECKED): Crossfire
    set rssi_channel = 12			# ELRS has LQ on 15 and dB on 16, CRSF on 12
    rxfail 12 s 1000				# Set RSSI/LQ to zero
  #$ OPTION END
#$ OPTION_GROUP END

# aux channel mapping
aux 0 0 0 1700 2100 0 0				# ARM, AUX 1 ON
aux 1 1 3 900 1300 0 0				# ANGLE, AUX 4 ON
aux 2 2 3 1300 1700 0 0				# HORIZON, AUX 4 NULL
aux 3 13 1 1300 1700 0 0			# BEEPER, AUX 2 MID
aux 4 26 5 1700 2100 0 0			# BLACKBOX, AUX 6 ON
aux 5 31 8 1700 2100 0 0			# BLACKBOX ERASE, AUX 9 ON
aux 6 35 1 1700 2100 0 0			# TURTLE MODE, AUX 2 LOW
aux 7 36 1 900 1300 0 0				# PREARM, AUX 2 LOW
aux 8 36 1 1700 2100 0 0			# PREARM, AUX 2 HIGH
aux 9 41 4 1700 2100 0 0			# USER 2 (CAM TOGGLE), AUX 5 ON

# adjustments
adjrange 0 0 2 900 2100 12 2 0 0		# Rateprofile selection

# input ranges
rxrange 0 988 2011
rxrange 1 988 2011
rxrange 2 988 2011
rxrange 3 988 2011

# arm angle limit off
set small_angle = 180
